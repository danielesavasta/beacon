# i18n — Astro nativo

## astro.config.mjs

```js
import { defineConfig } from 'astro/config';

export default defineConfig({
  site: 'https://username.github.io',
  base: '/repo-name',
  i18n: {
    defaultLocale: 'it',
    locales: ['it', 'en', 'fr'],
    routing: {
      prefixDefaultLocale: false, // /it/ → /, /en/ → /en/
    },
  },
});
```

## Helper utils.ts

```ts
// src/i18n/utils.ts
import it from './it.json';
import en from './en.json';
import fr from './fr.json';

const translations = { it, en, fr } as const;
type Locale = keyof typeof translations;

export function useTranslations(locale: Locale) {
  return function t(key: string): string {
    return key.split('.').reduce((obj: any, k) => obj?.[k], translations[locale]) ?? key;
  };
}

export function getLangFromUrl(url: URL): Locale {
  const [, lang] = url.pathname.split('/');
  if (lang in translations) return lang as Locale;
  return 'it';
}
```

## Uso in un componente .astro

```astro
---
import { getLangFromUrl, useTranslations } from '../i18n/utils';
const lang = getLangFromUrl(Astro.url);
const t = useTranslations(lang);
---

<h1>{t('hero.title')}</h1>
<nav aria-label={t('nav.label')}>
  <a href={`/${lang}/`}>{t('nav.home')}</a>
</nav>
```

## Language switcher (componente)

```astro
---
// src/components/LangSwitcher.astro
const { currentLang, currentPath } = Astro.props;
const locales = ['it', 'en', 'fr'];
---
<nav class="lang-switcher" aria-label="Selezione lingua">
  {locales.map(locale => (
    <a
      href={`/${locale}${currentPath}`}
      class:list={['lang-switcher__link', { 'is-active': locale === currentLang }]}
      aria-current={locale === currentLang ? 'page' : undefined}
      hreflang={locale}
    >
      {locale.toUpperCase()}
    </a>
  ))}
</nav>
```

## `<html lang>` nel layout base

```astro
---
// src/layouts/Base.astro
import { getLangFromUrl } from '../i18n/utils';
const lang = getLangFromUrl(Astro.url);
---
<!DOCTYPE html>
<html lang={lang}>
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <link rel="alternate" hreflang="it" href="/it/" />
    <link rel="alternate" hreflang="en" href="/en/" />
    <link rel="alternate" hreflang="fr" href="/fr/" />
    <slot name="head" />
  </head>
  <body>
    <a class="skip-link" href="#main-content">Vai al contenuto</a>
    <slot />
  </body>
</html>
```
